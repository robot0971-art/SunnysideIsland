var settings = UnityEditor.AddressableAssets.AddressableAssetSettingsDefaultObject.Settings;
var group = System.Linq.Enumerable.FirstOrDefault(settings.groups, x => x.Name == "Buildings");
if (group == null) return "No Buildings group";
var sb = new System.Text.StringBuilder();
foreach (var e in group.entries)
{
    sb.Append(e.address);
    sb.Append(" | ");
    sb.AppendLine(e.Asset.name);
}
return sb.ToString();
